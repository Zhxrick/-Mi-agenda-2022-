		    aumento = (ingreso_anual * 0.05)
		    MsgBox ("Total impuesto a pagar: ") & aumento

		    aumento = (ingreso_anual * 0.1)
		    MsgBox ("Total impuesto a pagar: ") & aumento

		    Else

			    aumento = (ingreso_anual * 0.15)
			    MsgBox ("Total impuesto a pagar: ") & aumento
			Else

				  MsgBox ("Total impuesto a pagar: ") & aumento

					aumento = (ingreso_anual * 0.25)
					MsgBox ("Total impuesto a pagar: ") & aumento
				    Else

				   End If

			End If

		    End If
